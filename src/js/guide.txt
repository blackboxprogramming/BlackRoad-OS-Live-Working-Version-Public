function getTimeSlots(){
  var d=new Date(),h=d.getHours(),m=d.getMinutes();
  var base=m>=30?30:0;
  function fmt(hr,mn){var ap=hr>=12?'PM':'AM';var h12=hr%12||12;return h12+':'+String(mn).padStart(2,'0')+' '+ap}
  var s1=fmt(h,base);
  var m2=base+30,h2=h;if(m2>=60){m2=0;h2++}
  var s2=fmt(h2%24,m2);
  var m3=m2+30,h3=h2;if(m3>=60){m3=0;h3++}
  var s3=fmt(h3%24,m3);
  return [s1,s2,s3];
}

var BLOCK_COLORS=['b1','b2','b3','b4','b5','b6'];

function renderGuide(){
  var el=document.getElementById('guide-body');
  var ch=S.channels.find(function(c){return c.ch===S.selectedCh});
  var times=getTimeSlots();
  var h='';

  // Preview area
  h+='<div class="guide-preview">';
  h+='<div class="guide-thumb">'+(ch?'<pre>'+THUMBS[ch.name]+'</pre>':'<img src="'+HERO_IMG+'">')+'</div>';
  h+='<div class="guide-info">';
  if(ch){
    h+='<div class="guide-info-ch">CH '+ch.ch+'</div>';
    h+='<div class="guide-info-name">'+ch.name+'</div>';
    h+='<div class="guide-info-desc">'+ch.desc+'</div>';
    h+='<div class="guide-info-meta">'+ch.domain+' / '+ch.agent+'</div>';
    if(ch.state==='live')h+='<div class="guide-info-live"><span class="live-dot"></span>LIVE'+(ch.w?' / '+ch.w+' watching':'')+'</div>';
  } else {
    h+='<div class="guide-info-name">BlackRoad Guide</div>';
    h+='<div class="guide-info-desc">Select a channel to preview</div>';
  }
  h+='</div></div>';

  // Time header
  h+='<div class="guide-time-header">';
  h+='<div class="guide-time-ch-col">CH</div>';
  times.forEach(function(t){h+='<div class="guide-time-slot">'+t+'</div>'});
  h+='</div>';

  // Channel grid rows
  S.channels.forEach(function(c,i){
    var sel=S.selectedCh===c.ch?' selected':'';
    var progs=PROGRAMS[c.name]||[c.name,c.name,'--'];
    var bci=((c.ch*7)%BLOCK_COLORS.length);

    h+='<div class="guide-row'+sel+'" data-ch="'+c.ch+'">';
    h+='<div class="guide-ch-num">'+c.ch+'</div>';

    // 3 time blocks
    for(var t=0;t<3;t++){
      var cls=BLOCK_COLORS[(bci+t)%BLOCK_COLORS.length];
      if(c.starred)cls='starred-block';
      else if(c.state==='live'&&t===0)cls='live-block';
      var prog=progs[t]||'--';
      var tag='';
      if(c.starred&&t===0)tag='<span class="blk-tag">\u2605 PRIME</span>';
      else if(c.state==='live'&&t===0)tag='<span class="blk-tag">LIVE</span>';
      h+='<div class="guide-block '+cls+'">'+prog+' '+tag+'</div>';
    }
    h+='</div>';
  });

  el.innerHTML=h;

  // Click handlers
  el.querySelectorAll('.guide-row').forEach(function(r){r.addEventListener('click',function(){
    S.selectedCh=parseInt(r.dataset.ch);localStorage.setItem('br-sel',S.selectedCh);renderGuide();updateViewer();
  })});
}
