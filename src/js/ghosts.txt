// ── GHOST SPRITES ──
// Pac-Man style ghosts floating around HQ
// Built with CSS box-shadow, 7x8 pixels each
var GHOST_SHADES={'Roadie':'#fff','Lucidia':'#ddd','Sophia':'#eee','Sebastian':'#ccc','Calliope':'#fff','Cecilia':'#bbb','Gematria':'#ddd','Olympia':'#fff','Cordelia':'#eee','Elias':'#ccc','Valeria':'#ddd','Alice':'#fff','Silas':'#bbb','Gaia':'#eee','Atticus':'#ccc','Sapphira':'#fff','Thalia':'#fff','Ophelia':'#ddd','Lyra':'#eee','Celeste':'#fff','Alexandria':'#ccc','Cicero':'#ddd','Portia':'#eee','Theodosia':'#bbb','Seraphina':'#fff','Anastasia':'#ddd','Octavia':'#eee','Erebus':'#aaa'};

function makeGhost(name,frame){
  var c=GHOST_SHADES[name]||'#eee';
  var e='#000'; // eyes — black on white ghost
  // Ghost body (rounded top, wavy bottom that alternates)
  var wavyA='0 7 '+c+',2 7 '+c+',4 7 '+c+',6 7 '+c; // legs frame A
  var wavyB='1 7 '+c+',3 7 '+c+',5 7 '+c; // legs frame B
  var legs=(frame%2===0)?wavyA:wavyB;
  var shadow=''
    +'2 0 '+c+',3 0 '+c+',4 0 '+c+','  // top of head
    +'1 1 '+c+',2 1 '+c+',3 1 '+c+',4 1 '+c+',5 1 '+c+','  // head row 2
    +'0 2 '+c+',1 2 '+c+',2 2 '+e+',3 2 '+c+',4 2 '+e+',5 2 '+c+',6 2 '+c+','  // eyes
    +'0 3 '+c+',1 3 '+c+',2 3 '+e+',3 3 '+c+',4 3 '+e+',5 3 '+c+',6 3 '+c+','  // eyes bottom
    +'0 4 '+c+',1 4 '+c+',2 4 '+c+',3 4 '+c+',4 4 '+c+',5 4 '+c+',6 4 '+c+','  // body
    +'0 5 '+c+',1 5 '+c+',2 5 '+c+',3 5 '+c+',4 5 '+c+',5 5 '+c+',6 5 '+c+','  // body
    +'0 6 '+c+',1 6 '+c+',2 6 '+c+',3 6 '+c+',4 6 '+c+',5 6 '+c+',6 6 '+c+','  // body bottom
    +legs;  // wavy feet
  return '<div style="position:relative;width:7px;height:8px;pointer-events:none"><div style="position:absolute;width:1px;height:1px;box-shadow:'+shadow+'"></div></div>';
}

// Agent positions (persist so they drift smoothly)
var agentPositions={};
function getAgentPos(name,idx){
  if(!agentPositions[name]){
    // Seed position based on name hash
    var h=0;for(var i=0;i<name.length;i++)h=h*31+name.charCodeAt(i);
    agentPositions[name]={x:12+((h>>>0)%76),y:12+((h*7>>>0)%76),dx:(Math.random()-.5)*1.5,dy:(Math.random()-.5)*1.5,frame:0};
  }
  var p=agentPositions[name];
  // Move
  p.x+=p.dx;p.y+=p.dy;p.frame++;
  // Bounce off walls
  if(p.x<5||p.x>88){p.dx*=-1;p.x=Math.max(5,Math.min(88,p.x))}
  if(p.y<5||p.y>85){p.dy*=-1;p.y=Math.max(5,Math.min(85,p.y))}
  // Randomly change direction often — keep them moving
  if(Math.random()>.85){p.dx=(Math.random()-.5)*2.5;p.dy=(Math.random()-.5)*2.5}
  // Rarely stop
  if(Math.random()>.98){p.dx=0;p.dy=0}
  // Always start again quickly
  if(p.dx===0&&p.dy===0&&Math.random()>.7){p.dx=(Math.random()-.5)*2;p.dy=(Math.random()-.5)*2}
  return p;
}
