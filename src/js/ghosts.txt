// Build a Minecraft-style character: 8 wide x 16 tall pixels
// Rendered at 2x scale for compact size
function makeAvatar(name){
  var cols=AGENT_COLORS[name]||['#888','#666','#1a1a1a'];
  var shirt=cols[0];
  var shirt_d=cols[1];
  var pants=cols[2];
  var hair=HAIR_COLORS[name]||'#333';
  var e='#fff'; // eye whites
  var ep='#222'; // eye pupils
  var _='transparent';

  // 8x16 pixel character (front view, Minecraft style)
  // Row format: [col0, col1, col2, col3, col4, col5, col6, col7]
  var rows=[
    // Row 0-1: Hair top
    [_,_,hair,hair,hair,hair,_,_],
    [_,hair,hair,hair,hair,hair,hair,_],
    // Row 2: Hair + forehead
    [_,hair,SKIN,SKIN,SKIN,SKIN,hair,_],
    // Row 3: Eyes
    [_,SKIN,e,ep,SKIN,e,ep,_],
    // Row 4: Nose/mouth
    [_,SKIN,SKIN,SKIN_D,SKIN,SKIN,SKIN,_],
    // Row 5: Chin
    [_,_,SKIN,SKIN,SKIN,SKIN,_,_],
    // Row 6-7: Neck + shirt top
    [_,_,_,SKIN,SKIN,_,_,_],
    [_,shirt,shirt,shirt,shirt,shirt,shirt,_],
    // Row 8-9: Shirt body + arms
    [shirt_d,shirt,shirt,shirt,shirt,shirt,shirt,shirt_d],
    [SKIN,shirt,shirt,shirt,shirt,shirt,shirt,SKIN],
    // Row 10: Shirt bottom + hands
    [SKIN,shirt_d,shirt,shirt,shirt,shirt,shirt_d,SKIN],
    [_,_,shirt_d,shirt_d,shirt_d,shirt_d,_,_],
    // Row 12-13: Pants
    [_,_,pants,pants,pants,pants,_,_],
    [_,_,pants,pants,pants,pants,_,_],
    // Row 14-15: Legs + shoes
    [_,_,pants,'#111',pants,'#111',_,_],
    [_,_,'#111','#111','#111','#111',_,_],
  ];

  // Build box-shadow string
  var shadows=[];
  for(var y=0;y<rows.length;y++){
    for(var x=0;x<rows[y].length;x++){
      if(rows[y][x]!==_){
        shadows.push(x+'px '+y+'px 0 '+rows[y][x]);
      }
    }
  }

  return '<div style="width:1px;height:1px;box-shadow:'+shadows.join(',')+';margin-right:7px;margin-bottom:15px"></div>';
}

// Agent grid positions
var agentPositions={};

function getAgentPos(name,idx,total){
  if(!agentPositions[name]){
    var cols=Math.ceil(Math.sqrt(total));
    var row=Math.floor(idx/cols);
    var col=idx%cols;
    var spacingX=84/(cols+1);
    var spacingY=72/(Math.ceil(total/cols)+1);
    agentPositions[name]={
      x:8+spacingX*(col+1),
      y:6+spacingY*(row+1),
      baseX:8+spacingX*(col+1),
      baseY:6+spacingY*(row+1),
      phase:Math.random()*Math.PI*2
    };
  }
  var p=agentPositions[name];
  // Gentle idle animation — slight bob
  p.phase+=0.03;
  p.x=p.baseX+Math.sin(p.phase)*0.8;
  p.y=p.baseY+Math.sin(p.phase*1.3)*0.6;
  return p;
}
