function makeAvatar(name,dir){
  var cols=AGENT_COLORS[name]||['#888','#666'];

  var sd=cols[1]; // shirt dark
  var h=HAIR_COLORS[name]||'#333';

  dir=dir||'down';

  var rows;
  if(dir==='down'){
    rows=[
      [_,_,h,h,h,h,_,_],
      [_,h,h,h,h,h,h,_],
      [_,h,h,h,h,h,h,_],
      [_,sk,sk,sk,sk,sk,sk,_],
      [_,s,s,s,s,s,s,_],
      [sk,s,s,sd,sd,s,s,sk],
      [_,sd,s,s,s,s,sd,_],
      [_,_,sd,_,_,sd,_,_],
    ];
  } else if(dir==='up'){
    rows=[
      [_,_,h,h,h,h,_,_],
      [_,h,h,h,h,h,h,_],
      [_,h,h,h,h,h,h,_],
      [_,h,h,h,h,h,h,_],

      [sk,s,s,sd,sd,s,s,sk],
      [_,sd,s,s,s,s,sd,_],
      [_,_,sd,_,_,sd,_,_],
    ];
  } else if(dir==='left'){
    rows=[
      [_,h,h,h,h,_,_,_],
      [h,h,h,h,h,h,_,_],
      [h,h,h,h,h,_,_,_],
      [sk,sk,sk,sk,_,_,_,_],

    ];

    rows=[
      [_,_,_,h,h,h,h,_],

}

// Agent positions + walking state
var agentPositions={};
var DIRS=['down','left','up','right'];

function getAgentPos(name,idx,total){
  if(!agentPositions[name]){
    // Seed positions spread across the office floor area (bottom 55%)
    var h=0;for(var i=0;i<name.length;i++)h=h*31+name.charCodeAt(i);
    var x=10+((h>>>0)%75);
    var y=48+((h*7>>>0)%42);
    agentPositions[name]={
      x:x,y:y,
      targetX:x,targetY:y,
      dir:'down',
      moving:false,
      waitTicks:Math.floor(Math.random()*20),
      step:0
    };
  }
  var p=agentPositions[name];

  // Walking AI — pick a destination, walk to it, pause, repeat
  if(p.waitTicks>0){
    p.waitTicks--;
    p.moving=false;
  } else if(!p.moving){
    // Pick new target within office floor bounds
    p.targetX=10+Math.random()*75;
    p.targetY=48+Math.random()*42;
    p.moving=true;
    p.step=0;
  }

  if(p.moving){
    var dx=p.targetX-p.x;
    var dy=p.targetY-p.y;
    var dist=Math.sqrt(dx*dx+dy*dy);
    if(dist<1.5){
      p.x=p.targetX;
      p.y=p.targetY;
      p.moving=false;
      p.waitTicks=20+Math.floor(Math.random()*40);
    } else {
      var speed=0.6;
      p.x+=dx/dist*speed;
      p.y+=dy/dist*speed;
      p.step++;
      // Face direction of movement
      if(Math.abs(dx)>Math.abs(dy)){
        p.dir=dx>0?'right':'left';
      } else {
        p.dir=dy>0?'down':'up';
      }
    }
  }
  return p;
}
