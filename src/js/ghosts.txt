// Clean circle avatars with initials + full names, positioned on a grid

// Agent colors — each agent gets a unique color from the brand gradient
var AGENT_COLORS={
  'Roadie':'#FF6B2B','Lucidia':'#CC00AA','Sophia':'#8844FF','Erebus':'#444',
  'Cecilia':'#4488FF','Gematria':'#00D4FF','Valeria':'#FF2255','Olympia':'#FF6B2B',
  'Cordelia':'#8844FF','Alice':'#4488FF','Sebastian':'#CC00AA','Calliope':'#00D4FF',
  'Elias':'#FF2255','Gaia':'#FF6B2B','Atticus':'#8844FF','Sapphira':'#4488FF',
  'Thalia':'#CC00AA','Ophelia':'#00D4FF','Lyra':'#FF2255','Celeste':'#FF6B2B',
  'Alexandria':'#8844FF','Cicero':'#4488FF','Portia':'#CC00AA','Theodosia':'#00D4FF',
  'Seraphina':'#FF2255','Anastasia':'#FF6B2B','Octavia':'#8844FF','Aria':'#4488FF'
};

function makeAvatar(name){
  var c=AGENT_COLORS[name]||'#737373';
  var initials=name.substring(0,1);
  return '<div style="width:28px;height:28px;border-radius:50%;background:'+c+';display:flex;align-items:center;justify-content:center;font-family:var(--sg);font-size:11px;font-weight:700;color:#000;flex-shrink:0;border:2px solid rgba(0,0,0,.3)">'+initials+'</div>';
}

// Agent grid positions — arranged in a neat layout, not random bouncing
var agentPositions={};
var agentFrame=0;

function getAgentPos(name,idx,total){
  if(!agentPositions[name]){
    // Place agents in a grid layout
    var cols=Math.ceil(Math.sqrt(total));
    var row=Math.floor(idx/cols);
    var col=idx%cols;
    var spacingX=88/(cols+1);
    var spacingY=80/(Math.ceil(total/cols)+1);
    agentPositions[name]={
      x:8+spacingX*(col+1),
      y:8+spacingY*(row+1),
      baseX:8+spacingX*(col+1),
      baseY:8+spacingY*(row+1),
      phase:Math.random()*Math.PI*2
    };
  }
  var p=agentPositions[name];
  // Gentle floating — breathe in place, not random chaos
  p.phase+=0.04;
  p.x=p.baseX+Math.sin(p.phase)*1.5;
  p.y=p.baseY+Math.cos(p.phase*0.7)*1.2;
  return p;
}
