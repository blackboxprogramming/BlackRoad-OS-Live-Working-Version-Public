  dir=dir||'down';

// Darken/lighten a hex color (handles both #RGB and #RRGGBB)
function shadeColor(color,percent){
  var hex=color.replace('#','');
  // Expand 3-digit to 6-digit
  if(hex.length===3)hex=hex[0]+hex[0]+hex[1]+hex[1]+hex[2]+hex[2];
  var num=parseInt(hex,16);
  var r=(num>>16)&0xFF;
  var g=(num>>8)&0xFF;
  var b=num&0xFF;
  r=Math.max(0,Math.min(255,Math.round(r*(1+percent))));
  g=Math.max(0,Math.min(255,Math.round(g*(1+percent))));
  b=Math.max(0,Math.min(255,Math.round(b*(1+percent))));
  return '#'+(0x1000000+r*0x10000+g*0x100+b).toString(16).slice(1);
}

// Agent positions + smooth walking state
var agentPositions={};
var worldAnimId=null;
var lastWorldTick=0;

function initAgentPos(name,idx,total){
  var h=0;for(var i=0;i<name.length;i++)h=h*31+name.charCodeAt(i);
  // Spread agents across the walkable floor area
  var x=12+((h>>>0)%72);
  var y=50+((h*7>>>0)%38);
  return {
    x:x,y:y,
    targetX:x,targetY:y,
    dir:'down',
    moving:false,
    waitMs:1000+Math.random()*3000,
    waitStart:performance.now(),
    step:0,
    speed:0.012+Math.random()*0.008 // slightly different speeds per agent
  };
}

// Points of interest agents walk toward (desks, conference, couch, server, etc.)
var POI=[
  // Desks
  {x:13,y:54},{x:33,y:54},{x:65,y:54},{x:85,y:54},
  // Conference table area

  {x:25,y:58},{x:75,y:75},{x:35,y:78},{x:60,y:82},{x:80,y:62},
];

function tickAgent(p,now){
  if(!p.moving){
    // Waiting — check if wait time is over
    if(now-p.waitStart>p.waitMs){
      // Pick a point of interest (sometimes random, sometimes POI)
      if(Math.random()<0.6){
        var poi=POI[Math.floor(Math.random()*POI.length)];
        p.targetX=poi.x+(Math.random()-0.5)*8;
        p.targetY=poi.y+(Math.random()-0.5)*6;
      } else {
        p.targetX=10+Math.random()*78;
        p.targetY=48+Math.random()*42;
      }
      // Clamp to walkable area
      p.targetX=Math.max(6,Math.min(94,p.targetX));
      p.targetY=Math.max(46,Math.min(92,p.targetY));
      p.moving=true;
      p.step=0;
    }
    return;
  }

  var dx=p.targetX-p.x;
  var dy=p.targetY-p.y;
  var dist=Math.sqrt(dx*dx+dy*dy);

  if(dist<0.5){
    p.x=p.targetX;
    p.y=p.targetY;
    p.moving=false;
    p.waitMs=2000+Math.random()*5000;
    p.waitStart=now;
  } else {
    p.x+=dx/dist*p.speed*16.67; // normalize to ~60fps feel
    p.y+=dy/dist*p.speed*16.67;
    p.step++;
    // Face direction of movement
    if(Math.abs(dx)>Math.abs(dy)){
      p.dir=dx>0?'right':'left';
    } else {
      p.dir=dy>0?'down':'up';
    }
  }
}

function getAgentPos(name,idx,total){
  if(!agentPositions[name]){
    agentPositions[name]=initAgentPos(name,idx,total);
  }
  return agentPositions[name];
}

// Smooth animation loop — runs independently of render interval
function startWorldLoop(){
  if(worldAnimId)return;
  function loop(now){
    // Tick all agents
    var names=Object.keys(agentPositions);
    for(var i=0;i<names.length;i++){
      tickAgent(agentPositions[names[i]],now);
    }
    // Update DOM elements by moving them (no innerHTML rebuild)
    names.forEach(function(name){
      var p=agentPositions[name];
      var el=document.getElementById('agent-'+name);
      if(el){
        el.style.left=p.x+'%';
        el.style.top=p.y+'%';
      }
      var elBig=document.getElementById('agent-big-'+name);
      if(elBig){
        elBig.style.left=p.x+'%';
        elBig.style.top=p.y+'%';
      }
    });
    // Rebuild sprites every 400ms to update facing direction
    if(now-lastWorldTick>400){
      lastWorldTick=now;
      rebuildWorldSprites();
    }
    worldAnimId=requestAnimationFrame(loop);
  }
  worldAnimId=requestAnimationFrame(loop);
}

function rebuildWorldSprites(){
  var el=document.getElementById('world-sprites');
  var elBig=document.getElementById('world-big-sprites');
  if(!el)return;
  var defaultAgents=['Roadie','Lucidia','Cecilia','Octavia','Olympia','Silas','Sebastian','Calliope','Aria','Thalia','Lyra','Sapphira','Seraphina','Alexandria','Theodosia','Sophia','Gematria','Portia','Atticus','Cicero','Valeria','Alice','Celeste','Elias','Ophelia','Gaia','Anastasia'];

  var html='';
  var htmlBig='';
  defaultAgents.forEach(function(name){
    var p=agentPositions[name];
    if(!p)return;
    var avatar=makeAvatar(name,p.dir);
    html+='<div id="agent-'+name+'" style="position:absolute;left:'+p.x+'%;top:'+p.y+'%;display:flex;flex-direction:column;align-items:center;transform:translate(-50%,-50%);pointer-events:none;will-change:left,top">'
      +'<div style="transform:scale(2);image-rendering:pixelated;transform-origin:center top">'+avatar+'</div>'
      +'<div style="font-family:var(--jb);font-size:8px;font-weight:600;color:#ccc;text-align:center;white-space:nowrap;text-shadow:0 1px 3px #000;margin-top:1px">'+name+'</div>'
      +'</div>';
    htmlBig+='<div id="agent-big-'+name+'" style="position:absolute;left:'+p.x+'%;top:'+p.y+'%;display:flex;flex-direction:column;align-items:center;transform:translate(-50%,-50%);pointer-events:none;will-change:left,top">'
